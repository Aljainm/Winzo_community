    <script>
        const pages = document.querySelectorAll('.page');
        const navButtons = document.querySelectorAll('.nav-button');
        const bottomNavCommunity = document.getElementById('nav-community');
        const groupTabs = document.querySelectorAll('.group-tab');
        const myGroupsContent = document.getElementById('content-my-groups');
        const discoverGroupsContent = document.getElementById('content-discover-groups');

        function showPage(pageId) {
            pages.forEach(page => {
                if (page.id === pageId) {
                    page.classList.remove('hidden');
                } else {
                    page.classList.add('hidden');
                }
            });
        }

        navButtons.forEach(button => {
            button.addEventListener('click', () => {
                const targetPageId = button.dataset.pageTarget;
                showPage(targetPageId);
            });
        });

        // Special handling for the main community nav button
        bottomNavCommunity.addEventListener('click', () => {
            showPage('page-hub');
        });
        
        // A quick way to navigate to the activity feed for demo purposes
        // In a real app, this would be from a notification or a specific button
        document.querySelector('span:nth-child(1) > span:nth-child(1)').addEventListener('click', () => {
             showPage('page-activity');
        });


        groupTabs.forEach(tab => {
            tab.addEventListener('click', () => {
                groupTabs.forEach(t => {
                    t.classList.remove('active-tab');
                    t.classList.add('inactive-tab');
                });
                tab.classList.add('active-tab');
                tab.classList.remove('inactive-tab');

                if (tab.id === 'tab-my-groups') {
                    myGroupsContent.classList.remove('hidden');
                    discoverGroupsContent.classList.add('hidden');
                } else {
                    myGroupsContent.classList.add('hidden');
                    discoverGroupsContent.classList.remove('hidden');
                }
            });
        });

        // Set initial page
        showPage('page-hub');
    </script>
